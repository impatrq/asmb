``` 
   src/
    │
    ├── Librerias/
    │    ├── DB/
    │    ├── Fuera de uso/

    │    ├── Libreria x.py
    │    └── ...
    ├── Pruebas/
    │    ├── Archivo.py
    │    └── ...
    └── Recursos/
        ├── Circuitos/
        │    ├── Circuito x/
        │    │    ├── Circuito.pdsprj
        │    │    └── ...
        │    └── .../
        └── Modelos/
```
